Union types
-- this is different from c++, but you can specify args to be the more than one type and then you'll have to check type or only use functions that are available to both types in the union.

function printID( id: number | string) {
	cnosole.log("Your ID is " + id);
}

printID(101);
printID("202"); // both of these will work
